    DEVELOPER ||--o{ GAME : "1 a muchos"
    PUBLISHER ||--o{ GAME : "1 a muchos"
    GENRE ||--o{ GAME : "1 a muchos"

    GAME {
        int GameId PK
        string Title "required, maxlength 200"
        string Description "maxlength 500"
        decimal Price "required"
        int ReleaseYear
        string CoverImageUrl
        int DeveloperId FK
        int PublisherId FK
        int GenreId FK
    }

    DEVELOPER {
        int DeveloperId PK
        string Name "required, maxlength 150"
        string Country
    }

    PUBLISHER {
        int PublisherId PK
        string Name "required, maxlength 150"
        string Country
    }

    GENRE {
        int GenreId PK
        string Name "required, maxlength 100"
    }
Cardinalidades: cada Developer, Publisher y Genre se relaciona con cero o muchos Game, mientras que cada Game pertenece a exactamente un Developer, un Publisher y un Genre.
